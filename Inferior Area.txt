#include<iostream>
#include<algorithm>

using namespace std;
 
int main() {
	
	float M[12][12];
	
	for(int i = 0; i < 12; i++) {
		for(int j = 0; j < 12 ; j++) {
			scanf("%f", &M[i][j]);
		}
	}
	
	char O;
	scanf("%c", &O);
	
	int sum = 0, cont = 0;
	int ini = 5, fin=6;

	for(int fila = 7; fila < 12; fila++){
		for(int col = ini; col <= fin; col++){
			sum+=M[fila][col];
			cont++;
		}
		ini--;
		fin++;
	}
	
	if(O == 'S'){
		printf("%.1f\n", sum);
	}	
	if(O == 'M') {
		printf("%.1f\n", sum/cont);
	}
	
	return 0;
}
